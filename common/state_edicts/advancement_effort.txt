edict_advancement_effort = {
	potential = {
		always = yes		#we support "potential" if modders want to have lots and just show some.
	}
	
	allow = {
		custom_trigger_tooltip = {
			tooltip = disable_state_edicts_active.tooltip
			OR = {
				NOT = { has_country_flag = state_edicts_everywhere }
				FROM = { has_state_edict = edict_advancement_effort }
			}
			OR = {
				NOT = { has_country_flag = disable_state_edicts_active }
				FROM = { has_new_edict = { edict = edict_advancement_effort } }
			}
			FROM = {
				NOT = {
					any_province_in_state = {
						OR = {
							has_province_modifier = edict_advancement_effort
							has_province_modifier = edict_centralization_effort
							has_province_modifier = edict_defensive_edict
							has_province_modifier = state_council_edict
							has_province_modifier = edict_of_absolutism
							has_province_modifier = edict_encourage_development
							has_province_modifier = edict_feudal_de_jure_law
							has_province_modifier = edict_promote_military_recruitment
							has_province_modifier = edict_protect_trade
							has_province_modifier = edict_religion_enforced
							has_province_modifier = edict_religious_unity
							has_province_modifier = silvar_ishu_edict
						}
					}
				}
			}
		}
	}
	
	modifier = {
		local_institution_spread = 0.33
	}
	
	color = { 50 173 192 }
	
	
	ai_will_do = {
		# has to be around at least 10 to be picked (depends on maintainance cost and general economy)
		factor = 10
		modifier = {
			factor = 0
			all_province_in_state = {
				NOT = { current_institution_growth = 0.5 }
			}
		}
		modifier = {
			factor = 5
			any_province_in_state = {
				is_capital = yes
			}
		}
		modifier = {
			factor = 3
			all_province_in_state = {
				AND = {
					current_institution_growth = 0.5
				}
			}
		}
	}
}
