namespace = edict_events

country_event = {
	id = edict_events.1
	title = edict_events.1.t
	desc = edict_events.1.d
	picture = CITY_VIEW_eventPicture
	hidden = yes
	
	is_triggered_only = yes
	
	trigger = {
		#always = no
		has_country_flag = disable_state_edicts_active
		variable_arithmetic_trigger = {
			export_to_variable = {
				which = thisMonth
				value = trigger_value:is_month
			}
			is_variable_equal = {
				which = thisMonth
				which = prevMonth
			}
		}
	}
	
	after = {
		clr_country_flag = disable_state_edicts_active
	}
	
	option = {
        name = edict_events.1.a
		
		#hidden
	}
}

province_event = {
	id = edict_events.2
	title = edict_events.2.t
	desc = edict_events.2.d
	picture = CITY_VIEW_eventPicture
	hidden = yes
	
	trigger = {
		owner = { ai = no }
		OR = {
			has_edict_flag = { edict = edict_advancement_effort }
			has_edict_flag = { edict = edict_centralization_effort }
			has_edict_flag = { edict = edict_defensive_edict }
			has_edict_flag = { edict = state_council_edict }
			has_edict_flag = { edict = edict_of_absolutism }
			has_edict_flag = { edict = edict_encourage_development }
			has_edict_flag = { edict = edict_feudal_de_jure_law }
			has_edict_flag = { edict = edict_promote_military_recruitment }
			has_edict_flag = { edict = edict_protect_trade }
			has_edict_flag = { edict = edict_religion_enforced }
			has_edict_flag = { edict = edict_religious_unity }
			has_edict_flag = { edict = silvar_ishu_edict }
			
			AND = {
				NOT = { has_state_edict = edict_advancement_effort }
				NOT = { has_state_edict = edict_centralization_effort }
				NOT = { has_state_edict = edict_defensive_edict }
				NOT = { has_state_edict = state_council_edict }
				NOT = { has_state_edict = edict_of_absolutism }
				NOT = { has_state_edict = edict_encourage_development }
				NOT = { has_state_edict = edict_feudal_de_jure_law }
				NOT = { has_state_edict = edict_promote_military_recruitment }
				NOT = { has_state_edict = edict_protect_trade }
				NOT = { has_state_edict = edict_religion_enforced }
				NOT = { has_state_edict = edict_religious_unity }
				NOT = { has_state_edict = silvar_ishu_edict }
				OR = {
					has_province_flag = edict_advancement_effort
					has_province_flag = edict_centralization_effort
					has_province_flag = edict_defensive_edict
					has_province_flag = state_council_edict
					has_province_flag = edict_of_absolutism
					has_province_flag = edict_encourage_development
					has_province_flag = edict_feudal_de_jure_law
					has_province_flag = edict_promote_military_recruitment
					has_province_flag = edict_protect_trade
					has_province_flag = edict_religion_enforced
					has_province_flag = edict_religious_unity
					has_province_flag = silvar_ishu_edict
				}
			}
		}
	}
	
	immediate = {
		set_edict_flag = { edict = edict_advancement_effort }
		set_edict_flag = { edict = edict_centralization_effort }
		set_edict_flag = { edict = edict_defensive_edict }
		set_edict_flag = { edict = state_council_edict }
		set_edict_flag = { edict = edict_of_absolutism }
		set_edict_flag = { edict = edict_encourage_development }
		set_edict_flag = { edict = edict_feudal_de_jure_law }
		set_edict_flag = { edict = edict_promote_military_recruitment }
		set_edict_flag = { edict = edict_protect_trade }
		set_edict_flag = { edict = edict_religion_enforced }
		set_edict_flag = { edict = edict_religious_unity }
		set_edict_flag = { edict = silvar_ishu_edict }
		
		log = "EDICT PROVINCE"
	}
	
	option = {
        name = edict_events.2.a
		
		#hidden
	}
}
